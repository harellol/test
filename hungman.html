import random def hangman(): # List of words to choose from word_list = ['python', 'hangman', 'challenge', 'programming', 'developer'] # Select a random word word = random.choice(word_list) guessed_letters = set() attempts = 6 word_set = set(word) print("Welcome to Hangman!") while attempts > 0 and word_set: # Display the current state of the word display_word = [letter if letter in guessed_letters else '_' for letter in word] print("Word:", ' '.join(display_word)) # Get user input guess = input("Guess a letter: ").lower() if guess in guessed_letters: print("You already guessed that letter.") continue elif len(guess) != 1 or not guess.isalpha(): print("Invalid input. Please enter a single letter.") continue guessed_letters.add(guess) if guess in word_set: word_set.remove(guess) print(f"Good guess! The letter '{guess}' is in the word.") else: attempts -= 1 print(f"Wrong guess. You have {attempts} attempts left.") # Check if the game is won if not word_set: print(f"Congratulations! You've guessed the word: {word}") return print(f"Game over! The word was: {word}") # Run the game if __name__ == "__main__": hangman()
